#!/bin/bash
procesar_archivo() {
    local archivo="$1"
    local extension="${archivo##*.}"
    
    case "$extension" in
        fasta|fa|fas)
            echo "üß¨ Procesando secuencia FASTA: $archivo"
            # Contar secuencias
            grep -c "^>" "$archivo" && echo "secuencias encontradas"
            ;;
        fastq|fq)
            echo "üìä Procesando reads FASTQ: $archivo"
            # QC r√°pido
            total_reads=$(($(wc -l < "$archivo") / 4))
            echo "Total de reads: $total_reads"
            ;;
        bam)
            echo "üóÇÔ∏è  Procesando alineamiento BAM: $archivo"
            samtools flagstat "$archivo" 2>/dev/null || echo "Archivo BAM corrupto"
            ;;
        vcf|vcf.gz)
            echo "üî¨ Procesando variantes VCF: $archivo"
            if [[ "$archivo" == *.gz ]]; then
                zcat "$archivo" | grep -v "^#" | wc -l
            else
                grep -v "^#" "$archivo" | wc -l
            fi
            echo "variantes encontradas"
            ;;
        bed)
            echo "üìç Procesando regiones BED: $archivo"
            wc -l < "$archivo" && echo "regiones definidas"
            ;;
        *)
            echo "‚ö†Ô∏è  Tipo de archivo desconocido: $archivo"
            echo "Extensiones soportadas: fasta, fastq, bam, vcf, bed"
            return 1
            ;;
    esac
}

# Uso
for archivo in data/*; do
    [[ -f "$archivo" ]] && procesar_archivo "$archivo"
done

    # 2. Alineamiento
    echo "üéØ Alineando contra referencia..."

        "raw_data/${sample}_R1.fastq.gz" \
        "raw_data/${sample}_R2.fastq.gz" | \

    # 3. Indexado
    echo "üìá Indexando BAM..."
    samtools index "alignments/${sample}.sorted.bam"
    
    echo "‚úÖ $sample completado"

echo "üéâ Pipeline finalizado para ${#samples[@]} muestras"

#!/bin/bash

# Crear estructura de directorios

    exit 1

    sample_name=$(basename "$sample_dir")
    
    echo "
üìä Procesando muestra: $sample_name"
    
    # Buscar archivos FASTQ
    r1_file=$(find "$sample_dir" -name "*R1*.fastq*" | head -1)
    r2_file=$(find "$sample_dir" -name "*R2*.fastq*" | head -1)
    
    if [[ -z "$r1_file" || -z "$r2_file" ]]; then
        echo "‚ö†Ô∏è  Archivos FASTQ incompletos para $sample_name, saltando..."
        continue
    fi
    
    # QC y alineamiento
    case "${r1_file##*.}" in
        gz)
            echo "üóúÔ∏è  Archivos comprimidos detectados"
            reads_count=$(($(zcat "$r1_file" | wc -l) / 4))
            ;;
        fastq)
            reads_count=$(($(wc -l < "$r1_file") / 4))
            ;;
        *)
            echo "‚ùå Formato no soportado: $r1_file"
            continue
            ;;
    esac
    
    echo "üìà Reads encontrados: $reads_count"
    
    # Solo procesar si hay suficientes reads
    if (( reads_count < 10000 )); then
        echo "‚ö†Ô∏è  Muy pocos reads ($reads_count), saltando $sample_name"
        continue
    fi
    
    # Alineamiento y llamada de variantes (simulado)
    bam_file="$results_dir/alignments/${sample_name}.bam"
    vcf_file="$results_dir/variants/${sample_name}.vcf"
    
    echo "üéØ Generando alineamiento..."
    echo "Simulated BAM for $sample_name" > "$bam_file"
    
    echo "üî¨ Llamando variantes..."
    echo "Simulated VCF for $sample_name" > "$vcf_file"
    
    echo "‚úÖ $sample_name procesado exitosamente"
done

echo "
üéâ Pipeline completado. Resultados en: $results_dir"

#!/bin/bash
# Funci√≥n para validar herramientas
check_dependencies() {
    local tools=("samtools" "bwa" "fastqc")
    local missing=()
    
    for tool in "${tools[@]}"; do
        if ! command -v "$tool" &> /dev/null; then
            missing+=("$tool")
        fi
    done
    
    if (( ${#missing[@]} > 0 )); then
        echo "‚ùå Herramientas faltantes: ${missing[*]}"
        echo "Instala con: conda install ${missing[*]}"
        exit 1
    fi
    
    echo "‚úÖ Todas las dependencias disponibles"
}

# Funci√≥n para progreso visual
show_progress() {
    local current=$1
    local total=$2
    local percent=$((current * 100 / total))
    local bar_length=20
    local filled=$((percent * bar_length / 100))
    
    printf "\rProgreso: ["
    printf "%*s" $filled | tr ' ' '‚ñà'
    printf "%*s" $((bar_length - filled)) | tr ' ' '‚ñë'
    printf "] %d%% (%d/%d)" $percent $current $total
}

# Uso en bucle
samples=(sample1 sample2 sample3 sample4 sample5)
total=${#samples[@]}

for i in "${!samples[@]}"; do
    show_progress $((i + 1)) $total
    # Simular procesamiento
    sleep 1
done
echo -e "\n‚úÖ Procesamiento completo"
